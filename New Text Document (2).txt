<!DOCTYPE html>
<html>
    <body>
        <h2>Random image</h2>
        <button type="button" onclick="getdog()">Get dog</button>
        <img src="" alt="Random dog image" id="doging" width="300">
        <script>
            function getdog() {
                fetch('https://dog.ceo/api/breeds/image/random')
                    .then(response => response.json())
                    .then(data => {
                        document.getElementById('doging').setAttribute('src', data.message);
                        console.log('status', data.status);
                    })
                    .catch(error => {
                        console.log('error', error);
                    });
            }
        </script>
    </body>
</html>
